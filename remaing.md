def moorti_nirnaya(transiting_planet, natal_moon_sign):
    # Based on transit sign from Moon
    house_from_moon = (transit_sign - natal_moon_sign) % 12 + 1
    
    moorti_map = {
        'swarna': [1, 6, 11],      # Gold = excellent
        'rajat': [2, 5, 9],        # Silver = good
        'tamra': [3, 7, 10],       # Copper = mixed
        'loha': [4, 8, 12]         # Iron = difficult
    }
    
    for quality, houses in moorti_map.items():
        if house_from_moon in houses:
            return quality
